//交互列
ll count(vi v,ll n){
  ll a=0,b=1;
  rep(i,n-1){
    if(v[i]!=v[i+1]) b++;
    else{
      chmax(a,b);
      b=1;
    }
  }
  chmax(a,b);
  return a;
}
//機械
vi inv_ill(vi v,ll a,ll b){
  rep(i,b-a+1){
    v[a+i]=1-v[a+i];
  }
  return v;
}

void solve() {
  ll n;
  cin>>n;
  vi a(n);
  rep(i,n) cin>>a[i];
  ll ans=0;
  rep(i,n-1){
    REP(j,i,n){
      chmax(ans,count(inv_ill(a,i,j),n));
    }
  }
  out(ans);
}
